#include<stdio.h>
 
#define PI 3.14
 
int main()
{
  float radius, area, diameter;
 
  printf("\n Please Enter the radius of a circle : ");
  scanf("%f",&radius);
 
  diameter = 2 * radius;
  area = PI * radius * radius; 
 
  printf("\n Diameter Of a Circle = %.2f\n", diameter);
  printf("\n Area Of a Circle = %.2f\n", area);
 
  return 0;
}
