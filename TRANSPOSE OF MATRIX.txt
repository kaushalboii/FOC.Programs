/*TRANSPOSE F THE MATRIX*/
#include<stdio.h>
#include<conio.h>
int main ()
{
	int m,n,i,j, matrix[10][10],transpose[10][10];
	printf("entre rows and columns :");
	scanf("%d%d",&m,&n);
	printf("enter elements of the matrix :\t");
	for (i=0;i<m;i++)
	for(j=0;j<n;j++)
	scanf("%d",&matrix[i][j]);
	for(i=0;i<m;i++)
	for(j=0;j<n;j++)
	transpose[i][j]=matrix[j][i];
	printf("transpose of the matrix\n\n :");
	for(i=0;i<m;i++)
	{
		for(j=0;j<n;j++)
		printf("%d\t",transpose[i][j]);
	}
	return 0;
	}
